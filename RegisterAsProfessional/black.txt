    <script>
        // Form submission handling with modal display
        document.getElementById('professionalForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            // Create hidden iframe for silent submission
            const iframe = document.createElement('iframe');
            iframe.style.display = 'none';
            iframe.name = 'hidden-form-submit';
            document.body.appendChild(iframe);
            
            // Set form target to iframe
            this.target = 'hidden-form-submit';
            
            // Show success modal
            document.getElementById('successModal').style.display = 'flex';
            
            // Submit the form
            this.submit();
            
            // Reset form after submission
            setTimeout(() => {
                this.reset();
                document.body.removeChild(iframe);
            }, 1000);
        });

        // Close modal handlers
        document.getElementById('closeModal').addEventListener('click', function() {
            document.getElementById('successModal').style.display = 'none';
        });

        document.getElementById('modalOkBtn').addEventListener('click', function() {
            document.getElementById('successModal').style.display = 'none';
        });

        // Close modal when clicking outside content
        window.addEventListener('click', function(event) {
            const modal = document.getElementById('successModal');
            if (event.target === modal) {
                modal.style.display = 'none';
            }
        });
    </script>
